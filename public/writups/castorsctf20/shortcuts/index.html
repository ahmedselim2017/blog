So if we can execute go file let’s execute ls -Rla with go and find the interesting files.

package main

import (
	"fmt"
	"os/exec"
)
func main() {
    cmd :=  exec.Command("/bin/ls", "-lahR", "/")
    stdout, err := cmd.Output()
    if err != nil {
        fmt.Println(err.Error())
    }
    fmt.Println(string(stdout))

}
